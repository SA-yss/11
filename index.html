create index if not exists warehouse_logs_item_date_idx on public.warehouse_logs(item_id, d desc); create or replace function public.approve_warehouse_request( p_request_id uuid, p_approved_by text default null ) returns table ( request_id uuid, item_id text, approved_at timestamptz, approved_by text, processed_log_key text, out_qty int, remaining_stock int ) language plpgsql security definer set search_path = public as $$ declare v_req public.warehouse_requests%rowtype; v_now timestamptz := now(); v_log_key text := 'reqlog_' || replace(gen_random_uuid()::text, '-', ''); v_remaining_stock int; begin select * into v_req from public.warehouse_requests where id = p_request_id for update; if not found then raise exception '요청을 찾을 수 없습니다.'; end if; if coalesce(v_req.status, 'pending') <> 'pending' then raise exception '이미 처리된 요청입니다. 현재 상태: %', v_req.status; end if; update public.warehouse_items set base_stock = base_stock - coalesce(v_req.qty, 0), updated_at = now() where id = v_req.item_id and base_stock >= coalesce(v_req.qty, 0) returning base_stock into v_remaining_stock; if v_remaining_stock is null then raise exception '재고가 부족합니다.'; end if; insert into public.warehouse_logs ( item_id, d, t, dept, person, qty, __key ) values ( v_req.item_id, (v_now at time zone 'Asia/Seoul')::date, '출고', coalesce(v_req.department, '-'), coalesce(v_req.requester_name, ''), coalesce(v_req.qty, 0), v_log_key ); update public.warehouse_requests set status = 'approved', approved_at = v_now, approved_by = p_approved_by, processed_log_key = v_log_key where id = p_request_id; return query select v_req.id, v_req.item_id, v_now, p_approved_by, v_log_key, coalesce(v_req.qty, 0), v_remaining_stock; end; $$; create or replace function public.reject_warehouse_request( p_request_id uuid, p_approved_by text default null ) returns table ( request_id uuid, rejected_at timestamptz, rejected_by text ) language plpgsql security definer set search_path = public as $$ declare v_req public.warehouse_requests%rowtype; v_now timestamptz := now(); begin select * into v_req from public.warehouse_requests where id = p_request_id for update; if not found then raise exception '요청을 찾을 수 없습니다.'; end if; if coalesce(v_req.status, 'pending') <> 'pending' then raise exception '이미 처리된 요청입니다. 현재 상태: %', v_req.status; end if; update public.warehouse_requests set status = 'rejected', approved_at = v_now, approved_by = p_approved_by, processed_log_key = null where id = p_request_id; return query select v_req.id, v_now, p_approved_by; end; $$; grant execute on function public.approve_warehouse_request(uuid, text) to anon, authenticated; grant execute on function public.reject_warehouse_request(uuid, text) to anon, authenticated;
